function doPost(e) { try { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); // Get form data const email = e.parameter.email || 'No email'; const timestamp = e.parameter.timestamp || new Date().toISOString(); const source = e.parameter.source || 'website'; // Add to sheet sheet.appendRow([email, timestamp, source]); // Return simple response return ContentService.createTextOutput('success'); } catch (error) { return ContentService.createTextOutput('error: ' + error.toString()); } } function doGet(e) { return ContentService.createTextOutput('Script is running. Use POST to submit emails.'); }
